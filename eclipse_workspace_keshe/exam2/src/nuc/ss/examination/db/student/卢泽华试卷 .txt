import java.util.Scanner;                 
class Rect {         
   private float a;
   private float b;
   public Rect() {
      Scanner sc = new Scanner(System.in);      
      System.out.println("长方形的长，宽是：");
      a = sc.nextFloat();                
      b = sc.nextFloat();
   }   
   public float getArea() {       
      return a*b;
   }
}


class Circle  //定义圆形类
{
   private float r;
   public Circle() {
      Scanner sc = new Scanner(System.in);
      System.out.println("圆形的半径是：");
      r = sc.nextFloat();         
   }   
    public float getArea() {
      return 3.14f*r*r; 
  }
}
  class Geometry {  //几何图形类的定义
 	 private Rect rect;   
  private Circle circle;  
   public Geometry(Rect rect,Circle circle) {
     this.rect = rect;
     this.circle = circle;
   }
   public float getArea() { //计算几何图形的面积 
      return rect.getArea() + circle.getArea();
   } 
 }          
public class GeometryArea {
  public static void main(String[] args) {
       System.out.println("请输入几何图形的信息");
       Rect rect = new Rect();  
       Circle circle = new Circle();   
       Geometry geometry = new Geometry(rect,circle); 
       System.out.println("几何图形的面积"+ geometry.getArea());      
  }
}
